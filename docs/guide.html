Getting Started

Set up http233-router-go in minutes and build REST APIs with middleware and route groups.

Installation

go get github.com/neko233-com/http233-router-go@v0.1.0

Basic Server

r := http233.New()
r.GET("/ping", func(c *http233.Context) {
    c.String(http.StatusOK, "pong")
})
http.ListenAndServe(":8080", r.Handler())

Route Parameters

r.GET("/users/:id", func(c *http233.Context) {
    id := c.Param("id")
    c.JSON(http.StatusOK, map[string]string{"id": id})
})

r.GET("/files/*path", func(c *http233.Context) {
    c.String(http.StatusOK, "path=%s", c.Param("path"))
})

Middleware

Middleware runs in registration order. Call c.Next() to continue the chain, or c.Abort() to stop.

r.Use(http233.Recovery())

r.Use(func(c *http233.Context) {
    start := time.Now()
    c.Next()
    log.Printf("%s %s %v", c.Request.Method, c.Request.URL.Path, time.Since(start))
})

r.GET("/api/data", func(c *http233.Context) {
    c.JSON(http.StatusOK, map[string]bool{"ok": true})
})

Route Groups

api := r.Group("/api/v1")
api.Use(authMiddleware)

users := api.Group("/users")
users.GET("", listUsers)
users.GET("/:id", getUser)
users.POST("", createUser)

admin := api.Group("/admin", adminOnly)
admin.GET("/stats", getStats)

Query & JSON Body

r.GET("/search", func(c *http233.Context) {
    q := c.Query("q")
    page := c.DefaultQuery("page", "1")
    c.String(http.StatusOK, "q=%s page=%s", q, page)
})

r.POST("/users", func(c *http233.Context) {
    var body struct{ Name string `json:"name"` }
    if err := c.BindJSON(&body); err != nil {
        c.AbortWithStatus(http.StatusBadRequest)
        return
    }
    c.JSON(http.StatusCreated, body)
})

Static Files

r.Static("/static", "./public")
r.StaticFile("/favicon.ico", "./public/favicon.ico")

405 & 404

r.HandleMethodNotAllowed()
r.RedirectTrailingSlash(true)

r.SetNotFoundHandler(func(c *http233.Context) {
    c.JSON(http.StatusNotFound, map[string]string{"error": "not found"})
})

Hot Reload (Development)

r.EnableHotReload("./public", "./templates")
r.SetHotReloadCallback(func(events []fsnotify.Event) {
    log.Println("assets changed, refresh cache")
})
